async function backendCheck() {
        if (privelege === "user") {
            try {
                const response = await Axios.post('/api/checkRegister', {username, password, email})
                if (response.data.error) {
                    appDispatch({type: "flashMessage", value: response.data.error})
                    return false
                }
                if (response.data.message) {
                    appDispatch({type: "flashMessage", value: response.data.message})
                    return true
                }
            } catch(e) {
                appDispatch({type: "flashMessage", value: "There was an issue connecting to the backend. Please try again later."})
                return false
            }
        }
        if (privelege === "admin") {
            appDispatch({type: "flashMessage", value: "Admin registration not yet supported."})
            return false
        }
    }

    async function backendInsertion() {
        try {
            const response = Axios.post('/api/register', {username, password, email})
            if (response.data.error) {
                appDispatch({type: "flashMessage", value: response.data.error})
                return false
            }
            if (response.data.message) { // Successful insertion occured here
                appDispatch({type: "flashMessage", value: response.data.message})
                return true
            }
        } catch(e) {
            appDispatch({type: "flashMessage", value: "There was an issue connecting to the backend. Please try again later."})
            return false
        }
    }
